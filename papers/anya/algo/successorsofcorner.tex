\begin{algorithmic}
\STATE {\bf input}: Graph $G$, corner $c = \langle x,y\rangle$, target $t$
\STATE $result := \{\}$
\STATE $left := x$
\WHILE{$\neg isObstacle([x,left-1])$}
  \STATE $left := left - 1$
\ENDWHILE
\STATE $right := x$
\WHILE{$\neg isObstacle([x,right])$}
  \STATE $right := right + 1$
\ENDWHILE
\STATE $i := \langle [(left,y+1),(right,y+1)], c\rangle$
\STATE {\bf return} $result$
\end{algorithmic}
