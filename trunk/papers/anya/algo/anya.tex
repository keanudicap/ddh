\begin{algorithmic}
\STATE {\bf input}: Graph $G$, source $s$, target $t$
\STATE $open := expand\_node(s)$
\LOOP
  \STATE $i := pop(open)$
  \IF{$i = t$}
    \STATE {\bf return} $i$
  \ENDIF
  \STATE $\langle I_\varepsilon, C\rangle := split\_epsilon(i)$
  \FORALL{$i' \in I_\varepsilon$}
    \STATE $open.insert(move(i'))$
  \ENDFOR
  \FORALL{$c \in C$}
    \STATE $open.insert(expand\_node(c))$
  \ENDFOR
\ENDLOOP
\end{algorithmic}
