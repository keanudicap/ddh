\newcommand{\creategridjps}[2]{%
  \draw[step=1] (0,0) grid (#1,#2);%
%
  \foreach \x in {1,...,#1}%
    \draw (\x - 0.5, -0.5) node {\x};%
%
  \foreach \y in {1,...,#2}%
    \draw (-0.2, \y - 0.5) node {\y};%
}%
\newcommand{\drawobstaclejps}[2]{%
\draw[fill=black] (#1,#2) rectangle +(-1,-1);
}%

  \draw (#1 - 0.5, #2 - 0.5) node {#3};
}
